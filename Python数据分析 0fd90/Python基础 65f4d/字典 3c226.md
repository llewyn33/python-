# 字典

获取value

```python
person = {'name': '吴签', 'age': 28}
print(person['name']) # 获取字典中不存在的key时，报错
print(person.get('name')) # 获取字典中不存在的key时，返回none
```

修改

```python
person = {'name': '张三', 'age': 18}
person['name'] = '法外狂徒'
```

添加 

```python
person = {'name': '老马'}
person['age'] = 18 # 若key值存在则为修改
print(person)
```

删除

```python
# del
person = {'name': '老马','age': 18}
del person['age']
print(person)

#pop 删除的同时返回被删的值，并删除键
ret = d1.pop('键名')

# clear
person = {'name': '老马','age': 18}
person.clear()
print(person)
```

遍历

```python
# 遍历字典的key
person = {'name': '老马','age': 18,'sex': '男'}
for key in person.keys():
    print(key)

#遍历字典的value
person = {'name': '老马','age': 18,'sex': '男'}
for value in person.values():
    print(value)

#遍历字典的key,value
person = {'name': '老马','age': 18,'sex': '男'}
for key,value in person.items():
    print(key,value)

#遍历字典的项/元素
person = {'name': '老马','age': 18,'sex': '男'}
for item in person.items():
    print(item)
```

合并

```python
# 对于已存在的键，值会被覆盖

d1 = {}
d2 = {}
d1 = {'a' : '1', 'b' : '2'}
d2 = {'c' : '3', 'd' : '4'}

d1.update(d2)
print(d1)
```

序列转字典

```python
a_list = ['foo', 'bar', 'baz']
b_list = ['one', 'two', 'three']

mapping = dict(zip(a_list,b_list))
print(mapping)
```

默认值

```python
from collections import defaultdict

words = ['apple', 'bat', 'bar', 'atom', 'book']
by_letter = defaultdict(list)
for word in words:
    by_letter[word[0]].append(word)

print(by_letter)

out:{'a': ['apple', 'atom'], 'b': ['bat', 'bar', 'book']})
```
